#include <vector>
#include <algorithm>
using namespace std;
class Student {
private:
    string name;
    int age;
public:
    Student(string n, int a) : name(n), age(a) {}
    void display() {
        cout << "Name: " << name << endl;
    }
};
int main() {
    vector<int> numbers = {1, 2, 3, 4, 5};
    for(auto num : numbers) {
        cout << num << " ";
    }
    return 0;
}
